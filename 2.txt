DEKLARASI:
n : number

DESKRIPSI:

START

INPUT n

FOR 0 TO n DO
  IF n MOD 3 = 0 THEN
    DISPLAY "Fizz"
  ESLE
    IF n MOD 5 = 0 THEN
      DISPLAY "Buzz"
    ELSE
      IF n MOD 3 = 0 AND n MOD 5 = 0 THEN
        DISPLAY "FizzBuzz"
      ELSE
        DISPLAY n
      ENDIF
    ENDIF
  ENDIF
ENDFOR

END
